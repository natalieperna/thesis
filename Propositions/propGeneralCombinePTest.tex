\begin{proposition}
\label{prop:general_combine_P_test}	
	Let $P$ and $Q$ be a relations that satisfy $\lnotation{\exists}{X}{}{\RAcompose{X}{P} = Q}$, then $\RAcompose{X}{\big(P \opA \lnotation{\opA}{i}{1 \le i \le n}{R_i}\big)} = Q$  has a solution for any relations $R_i$ and any monoid $\monoid{\Rel}{\opA}{\one{\opA}}$.
	
\begin{proof}~
	
	Let $p(n) \deq \RAcompose{X}{(P \opA \lnotation{\opA}{i}{1 \le i \le n}{R_i})} = Q$ has a solution.\\
	
	\underline{Base Step}: 
	\Beginproof 
		\pred{$p(1)$}
		$\mIff$ \com{Definition of $p(n)$}
		\pred{$\RAcompose{X}{(P \opA \lnotation{\opA}{i}{1 \le i \le 1}{R_i})} = Q$ has a solution}
		$\mIff$ \com{One-point Rule}
		\pred{$\RAcompose{X}{(P \opA R_1)} = Q$ has a solution}
		$\mIff$ \com{Hypothesis: $\RAcompose{X}{P} = Q$ has a solution \!\!\!\!\!\hsep\!\!\!\!\! Proposition~\ref{prop:combine_P_test}}
		\pred{$\true$}
	\Endproof
	
	\underline{Inductive Step}:
	\Beginproof 
		\pred{$p(n+1)$}
		$\mIff$ \com{Definition of $p(n)$}
		\pred{$\RAcompose{X}{(P \opA \lnotation{\opA}{i}{1 \le i \le n+1}{R_i})} = Q$ has a solution}
		$\mIff$ \com{Range Split}
		\pred{$\RAcompose{X}{(P \opA \lnotation{\opA}{i}{1 \le i \le n}{R_i} \opA \lnotation{\opA}{i}{n+1 \le i \le n+1}{R_i})} = Q$ has a solution}
		$\mIff$ \com{One-point Rule}
		\pred{$\RAcompose{X}{(P \opA \lnotation{\opA}{i}{1 \le i \le n}{R_i} \opA R_{n+1})} = Q$ has a solution}
		$\mIff$ \com{$P' = P \opA \lnotation{\opA}{i}{1 \le i \le n}{R_i}$ \hsep Induction Hypothesis}
		\pred{$\RAcompose{X}{(P' \opA R_{n+1})} = Q$ has a solution}
		$\mIff$ \com{Hypothesis: $\RAcompose{X}{P} = Q$ has a solution \!\!\!\!\!\hsep\!\!\!\!\! Proposition~\ref{prop:combine_P_test}}
		\pred{$\true$}
	\Endproof
	
\end{proof}

\end{proposition}
