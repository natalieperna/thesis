\begin{proposition}
\label{prop:combine_P_test}	
	\mynote{1}{THIS DOES NOT MAKE SENSE, EXCEPT THAT IT CAN BE A DIRECT APPLICATION OF PROPOSITION~\ref{prop:test}.}
	
	Let $P$ and $Q$ be a relations that satisfy $\lnotation{\exists}{X}{}{\RAcompose{X}{P} = Q}$, then $\RAcompose{X}{(P \opA R)} = Q$ has a solution for any relation $R$ and any operation $\opA$.\\
	
\begin{proof}~

	\Beginproof 
		\pred{$\RAcompose{X}{(P \opA R)} = Q$ has a solution}
		$\mIff$ \com{Formalization}
		\pred{$\lnotation{\exists}{X}{}{\RAcompose{X}{(P \opA R)} = Q}$}
		$\mIff$ \com{Antisymmetry}
		\pred{$\lnotation{\exists}{X}{}{\RAcompose{X}{(P \opA R)} \STleq Q \nAnd Q \STleq \RAcompose{X}{(P \opA R)}}$}
		$\mIff$ \com{Proposition~\ref{prop:residue}(i)}
		\pred{$\lnotation{\exists}{X}{}{X \STleq \RAleftresidual{Q}{(P \opA R)} \nAnd Q \STleq \RAcompose{X}{(P \opA R)}}$}
		$\mIff$ \com{Definition of $\STleq$}
		\pred{$\lnotation{\exists}{X}{}{\big(X = \RAleftresidual{Q}{(P \opA R)} \Ors X \STlt \RAleftresidual{Q}{(P \opA R)} \big) \nAnd Q \STleq \RAcompose{X}{(P \opA R)}}$}
		$\mIff$ \com{Distributivity of $\exists$ over $\Or$}
		\pred{$\lnotation{\exists}{X}{}{X = \RAleftresidual{Q}{(P \opA R)} \nAnd Q \STleq \RAcompose{X}{(P \opA R)}} \quad \Ors \\ \lnotation{\exists}{X}{}{X \STlt \RAleftresidual{Q}{(P \opA R)} \nAnd Q \STleq \RAcompose{X}{(P \opA R)}}$}
		$\mIff$ \com{Trading}
		\pred{$\lnotation{\exists}{X}{X = \RAleftresidual{Q}{(P \opA R)}}{Q \STleq \RAcompose{X}{(P \opA R)}} \quad \Ors \\ \lnotation{\exists}{X}{}{X \STlt \RAleftresidual{Q}{(P \opA R)} \nAnd Q \STleq \RAcompose{X}{(P \opA R)}}$}
		$\mIff$ \com{One-Point Axiom}
		\pred{$Q \STleq \RAcompose{X}{(P \opA R)}\subst{X}{\RAleftresidual{Q}{(P \opA R)}} \quad \Ors \\ \lnotation{\exists}{X}{}{X \STlt \RAleftresidual{Q}{(P \opA R)} \nAnd Q \STleq \RAcompose{X}{(P \opA R)}}$}
		$\mIff$ \com{Substitution}
		\pred{$Q \STleq \RAcompose{(\RAleftresidual{Q}{(P \opA R)})}{(P \opA R)} \Ors \lnotation{\exists}{X}{}{X \STlt \RAleftresidual{Q}{(P \opA R)} \nAnd Q \STleq \RAcompose{X}{(P \opA R)}}$}
		$\mIff$ \com{\dots Proposition~\ref{prop:test} (\ie $Q \STleq \RAcompose{(\RAleftresidual{Q}{(P \opA R)})}{(P \opA R)}$)}
		\pred{$\true \Ors \lnotation{\exists}{X}{}{X \STlt \RAleftresidual{Q}{(P \opA R)} \nAnd Q \STleq \RAcompose{X}{(P \opA R)}}$}
		$\mIff$ \com{Zero of $\Or$}
		\pred{$\true$}
	\Endproof
	
\end{proof}

\end{proposition}
