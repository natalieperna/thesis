\begin{proposition}
\label{prop:test}
	$\RAcompose{X}{P} = Q$ has a solution if and only if $Q = \RAcompose{(\RAleftresidual{Q}{P})}{P}$
	
\begin{proof}~
	
	$(\mImpl) \quad Q = \RAcompose{(\RAleftresidual{Q}{P})}{P} \mImp \RAcompose{X}{P} = Q$ has a solution \newline
	
	\Beginproof 
		\pred{$\RAcompose{X}{P} = Q$ has a solution}
		$\mIff$ \com{Formalization}
		\pred{$\lnotation{\exists}{X}{}{\RAcompose{X}{P} = Q}$}
		$\mIff$ \com{Antisymmetry}
		\pred{$\lnotation{\exists}{X}{}{\RAcompose{X}{P} \STleq Q \nAnd Q \STleq \RAcompose{X}{P}}$}
		$\mIff$ \com{Proposition~\ref{prop:residue}(i)}
		\pred{$\lnotation{\exists}{X}{}{X \STleq \RAleftresidual{Q}{P} \nAnd Q \STleq \RAcompose{X}{P}}$}
		$\mIff$ \com{Definition of $\STleq$}
		\pred{$\lnotation{\exists}{X}{}{\big(X = \RAleftresidual{Q}{P} \Ors X \STlt \RAleftresidual{Q}{P} \big) \nAnd Q \STleq \RAcompose{X}{P}}$}
		$\mIff$ \com{Distributivity of $\exists$ over $\Or$}
		\pred{$\lnotation{\exists}{X}{}{X = \RAleftresidual{Q}{P} \nAnd Q \STleq \RAcompose{X}{P}} \Ors \lnotation{\exists}{X}{}{X \STlt \RAleftresidual{Q}{P} \nAnd Q \STleq \RAcompose{X}{P}}$}
		$\mIff$ \com{Trading}
		\pred{$\lnotation{\exists}{X}{X = \RAleftresidual{Q}{P}}{Q \STleq \RAcompose{X}{P}} \Ors \lnotation{\exists}{X}{}{X \STlt \RAleftresidual{Q}{P} \nAnd Q \STleq \RAcompose{X}{P}}$}
		$\mIff$ \com{One-Point Axiom}
		\pred{$Q \STleq \RAcompose{X}{P}\subst{X}{\RAleftresidual{Q}{P}} \Ors \lnotation{\exists}{X}{}{X \STlt \RAleftresidual{Q}{P} \nAnd Q \STleq \RAcompose{X}{P}}$}
		$\mIff$ \com{Substitution}
		\pred{$Q \STleq \RAcompose{(\RAleftresidual{Q}{P})}{P} \Ors \lnotation{\exists}{X}{}{X \STlt \RAleftresidual{Q}{P} \nAnd Q \STleq \RAcompose{X}{P}}$}
		$\mIff$ \com{Hypothesis: $Q = \RAcompose{(\RAleftresidual{Q}{P})}{P}$}
		\pred{$\true \Ors \lnotation{\exists}{X}{}{X \STlt \RAleftresidual{Q}{P} \nAnd Q \STleq \RAcompose{X}{P}}$}
		$\mIff$ \com{Zero of $\Or$}
		\pred{$\true$}
	\Endproof
	
	\noindent $(\mImp) \quad \RAcompose{X}{P} = Q$ has a solution $ \mImp Q = \RAcompose{(\RAleftresidual{Q}{P})}{P}$ \newline
	
	\Beginproof 
		\pred{$\RAcompose{X}{P} = Q$ has a solution}
		$\mIff$ \com{Formalization}
		\pred{$\lnotation{\exists}{X}{}{\RAcompose{X}{P} = Q}$}
		$\mIff$ \com{Antisymmetry}
		\pred{$\lnotation{\exists}{X}{}{\RAcompose{X}{P} \STleq Q \nAnd Q \STleq \RAcompose{X}{P}}$}
		$\mIff$ \com{Proposition~\ref{prop:residue}(i)}
		\pred{$\lnotation{\exists}{X}{}{X \STleq \RAleftresidual{Q}{P} \nAnd Q \STleq \RAcompose{X}{P}}$}
		$\mImp$ \com{Isotony of $\RAcomp$}
		\pred{$\lnotation{\exists}{X}{}{\RAcompose{X}{P} \STleq \RAcompose{(\RAleftresidual{Q}{P})}{P} \nAnd Q \STleq \RAcompose{X}{P}}$}
		$\mImp$ \com{Transitivity of $\STleq$ \hsep Idempotency of $\AAnd$}
		\pred{$\lnotation{\exists}{X}{}{Q \STleq \RAcompose{X}{P} \STleq \RAcompose{(\RAleftresidual{Q}{P})}{P} \nAnd Q \STleq \RAcompose{(\RAleftresidual{Q}{P})}{P}}$}
		$\mIff$ \com{Distributivity of $\AAnd$ over $\exists$}
		\pred{$Q \STleq \RAcompose{(\RAleftresidual{Q}{P})}{P} \nAnd \lnotation{\exists}{X}{}{Q \STleq \RAcompose{X}{P} \STleq \RAcompose{(\RAleftresidual{Q}{P})}{P}}$}
		$\mImp$ \com{Weakening}
		\pred{$Q \STleq \RAcompose{(\RAleftresidual{Q}{P})}{P}$}
		$\mImp$ \com{Proposition~\ref{prop:residueproperties}(vii)}
		\pred{$Q \STleq \RAcompose{(\RAleftresidual{Q}{P})}{P} \STleq Q$}
		$\mIff$ \com{Antisymmetry}
		\pred{$Q = \RAcompose{(\RAleftresidual{Q}{P})}{P}$}
	\Endproof
	
\end{proof}

\end{proposition}
