		Compute(p, q, r)
		    DECL test1, test2, res
		    BEG  test1 = incl(p,q*(r^ & (q\p)));
		         test2 = incl(q,p*(r  & (p\q)));
		         IF test1 & test2
		            THEN res = r & syq(p,q)
		            ELSE IF test1
		               THEN res = r & (q\p)^
		               ELSE IF test2
		                  THEN res = r & (p\q)
		                  ELSE res = false 
		               FI 
		            FI 
		         FI       
		         RETURN res
		    END.
