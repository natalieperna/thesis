\begin{corollary}
\label{cor:compute_bij}
	Let $P$ be a bijection containing confidential information. Let $Q$ be a bijection representing an information observed on the monitored communication channel. Let $R$ be a filtering relation allowing for the selection of particular elements of the relations $P$ and $Q$. The confidential information contained in $P$ is being leaked as that represented by $Q$ via the abstraction relation $X = R \STmeet (\RArightresidual{P}{Q})$ if and only if  $\RArightresidual{P}{Q} = \RAconverse{(\RArightresidual{Q}{P})}$
	
\begin{proof}~
	
	According to the problem formulation illustrated by Figure~\ref{fig:commutativity}, we need to find solutions to Equation~\ref{eqn:commute1} and Equation~\ref{eqn:commute2}. \newline
	
	\Beginproof 
		\pred{$X = R \STmeet (\RArightresidual{P}{Q})$}
		$\mIff$ \com{Corollary~\ref{cor:compute}}
		\pred{$P \STleq \RAcompose{Q}{\big(\RAconverse{R} \STmeet (\RArightresidual{Q}{P})\big)} \nAnd Q \STleq \RAcompose{P}{\big(R \STmeet (\RArightresidual{P}{Q})\big)}$}
		$\mIff$ \com{Converse both sides \!\!\!\!\!\hsep\!\!\!\!\! Proposition~\ref{prop:relationproperties}(ii) \!\!\!\!\!\hsep\!\!\!\!\! Proposition~\ref{prop:relationproperties}(vi) \!\!\!\!\!\hsep\!\!\!\!\! Proposition~\ref{prop:relationproperties}(vii) \!\!\!\!\!\hsep\!\!\!\!\! Proposition~\ref{prop:residueproperties}(ii)}
		\pred{$\RAconverse{P} \STleq \RAcompose{\big(R \STmeet (\RAleftresidual{\RAconverse{P}}{\RAconverse{Q}})\big)}{\RAconverse{Q}} \nAnd \RAconverse{Q} \STleq \RAcompose{\big(\RAconverse{R} \STmeet (\RAleftresidual{\RAconverse{Q}}{\RAconverse{P}})\big)}{\RAconverse{P}}$}
		$\mIff$ \com{Proposition~\ref{prop:compute}}
		\pred{$\lnotation{\exists}{X}{\!\!X = R \!\STmeet\! (\RAleftresidual{\RAconverse{P}}{\RAconverse{Q}}) \!\nAnd\! \RAconverse{X} = \RAconverse{R} \!\STmeet\! (\RAleftresidual{\RAconverse{Q}}{\RAconverse{P}})}{\RAcompose{X}{\RAconverse{Q}} = \RAconverse{P} \!\nAnd\! \RAcompose{\RAconverse{X}}{\RAconverse{P}} = \RAconverse{Q}}$}
		$\mIff$ \com{Converse both sides \hsep Proposition~\ref{prop:relationproperties}(ii)}
		\pred{$\lnotation{\exists}{X}{X = R \!\STmeet\! (\RAleftresidual{\RAconverse{P}}{\RAconverse{Q}}) \!\nAnd\! X = R \!\STmeet\! (\RArightresidual{P}{Q})}{\RAcompose{X}{\RAconverse{Q}} = \RAconverse{P} \!\nAnd\! \RAcompose{\RAconverse{X}}{\RAconverse{P}} = \RAconverse{Q}}$} 
		$\mIff$ \com{Proposition~\ref{prop:residueproperties}(ii)}
		\pred{$\lnotation{\exists}{X}{X = R \STmeet \RAconverse{(\RArightresidual{Q}{P})} \nAnd X = R \STmeet (\RArightresidual{P}{Q})}{\RAcompose{X}{\RAconverse{Q}} = \RAconverse{P} \!\nAnd\! \RAcompose{\RAconverse{X}}{\RAconverse{P}} = \RAconverse{Q}}$}
		$\mIff$ \com{$P$ is a bijection \hsep $Q$ is a bijection \hsep $P \STleq \RAcompose{Q}{(\RArightresidual{Q}{P})}$ \hsep \\ $Q \STleq \RAcompose{P}{(\RArightresidual{P}{Q})}$ \hsep Proposition~\ref{prop:test_simplificaiton}(v)}
		\pred{$\lnotation{\exists}{X}{X = R \STmeet (\RArightresidual{P}{Q}) \nAnd X = R \STmeet (\RArightresidual{P}{Q})}{\RAcompose{X}{\RAconverse{Q}} = \RAconverse{P} \nAnd \RAcompose{\RAconverse{X}}{\RAconverse{P}} = \RAconverse{Q}}$}
		$\mIff$ \com{Idempotency of $\AAnd$}
		\pred{$\lnotation{\exists}{X}{X = R \STmeet (\RArightresidual{P}{Q})}{\RAcompose{X}{\RAconverse{Q}} = \RAconverse{P} \nAnd \RAcompose{\RAconverse{X}}{\RAconverse{P}} = \RAconverse{Q}}$}
		$\mIff$ \com{Proposition~\ref{prop:residue}(i)}
		\pred{$\lnotation{\exists}{X}{X = R \STmeet (\RArightresidual{P}{Q})}{X = \RAleftresidual{\RAconverse{P}}{\RAconverse{Q}} \nAnd \RAconverse{X} = \RAleftresidual{\RAconverse{Q}}{\RAconverse{P}}}$}
		$\mIff$ \com{Converse both sides \hsep Proposition~\ref{prop:residueproperties}(i)}
		\pred{$\lnotation{\exists}{X}{X = R \STmeet (\RArightresidual{P}{Q})}{X = \RAconverse{(\RArightresidual{Q}{P})} \nAnd X = \RArightresidual{P}{Q}}$}
		$\mIff$ \com{Converse both sides \hsep Proposition~\ref{prop:residueproperties}(i)}
		\pred{$\lnotation{\exists}{X}{X = R \STmeet (\RArightresidual{P}{Q})}{\RAconverse{(\RArightresidual{Q}{P})} = \RArightresidual{P}{Q} \nAnd X = \RArightresidual{P}{Q}}$}
		$\mIff$ \com{One-Point Axiom \hsep $R = \RAtop$ \hsep Identity of $\STmeet$}
		\pred{$\big(\RAconverse{(\RArightresidual{Q}{P})} = \RArightresidual{P}{Q} \nAnd X = \RArightresidual{P}{Q}\big)\subst{X}{\RArightresidual{P}{Q}}$}
		$\mIff$ \com{Substitution}
		\pred{$\RAconverse{(\RArightresidual{Q}{P})} = \RArightresidual{P}{Q} \nAnd \RArightresidual{P}{Q} = \RArightresidual{P}{Q}$}
		$\mIff$ \com{Reflexivity of $=$ \hsep Identity of $\AAnd$}
		\pred{$\RAconverse{(\RArightresidual{Q}{P})} = \RArightresidual{P}{Q}$}
	\Endproof

\end{proof}

\end{corollary}
