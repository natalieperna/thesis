\begin{corollary}
\label{cor:modulate}
	Let $P$ be a relation containing confidential information. Let $M$ be a total and injective relation that modulates the confidential information in some way such that the modulated confidential information is represented by $(\RAcompose{P}{M})$. Let $Q$ be a relation representing an information observed on the monitored communication channel. If the confidential information contained in $P$ is being leaked as that represented by $Q$ then any modulation of the confidential information contained in $P$ is also being leaked as that represented by $Q$, \ie $\lnotation{\exists}{X}{}{\RAcompose{P}{X} = Q} \mImp \lnotation{\exists}{Y}{}{\RAcompose{\RAcompose{P}{M}}{Y} = Q}$. \newline
	
	\begin{proof}~

		\Beginproof 
			\pred{$\lnotation{\exists}{X}{}{\RAcompose{P}{X} = Q}$}
			$\mIff$ \com{Converse both sides \hsep Proposition~\ref{prop:relationproperties}(vii)}
			\pred{$\lnotation{\exists}{X}{}{\RAcompose{\RAconverse{X}}{\RAconverse{P}} = \RAconverse{Q}}$}
			$\mIff$ \com{Proposition~\ref{prop:test}}
			\pred{$\RAconverse{Q} = \RAcompose{(\RAleftresidual{\RAconverse{Q}}{\RAconverse{P}})}{\RAconverse{P}}$}
			$\mIff$ \com{Converse both sides \hsep Proposition~\ref{prop:relationproperties}(ii) \hsep \\ Proposition~\ref{prop:relationproperties}(vii) \hsep Proposition~\ref{prop:residueproperties}(i)}
			\pred{$Q = \RAcompose{P}{(\RArightresidual{P}{Q})}$}
			$\mIff$ \com{Definition~\ref{def:residue}(i)}
			\pred{$Q = \RAcompose{P}{\RAcomplement{\RAcompose{\RAconverse{P}}{\RAcomplement{Q}}}}$}
			$\mImp$ \com{$M \text{ is total} \mIff \RAid \STleq \RAcompose{M}{\RAconverse{M}}$}
			\pred{$Q = \RAcompose{P}{\RAcomplement{\RAcompose{\RAconverse{P}}{\RAcomplement{Q}}}} \STleq \RAcompose{\RAcompose{P}{\RAcompose{M}{\RAconverse{M}}}}{\RAcomplement{\RAcompose{\RAconverse{P}}{\RAcomplement{Q}}}}$}
			$\mIff$ \com{$\RAconverse{M} \text{ is deterministic} \mIff \RAconverse{M} \text{ is injective}$}
			\pred{$Q = \RAcompose{P}{\RAcomplement{\RAcompose{\RAconverse{P}}{\RAcomplement{Q}}}} \STleq \RAcompose{\RAcompose{P}{M}}{\RAcomplement{\RAcompose{\RAcompose{\RAconverse{M}}{\RAconverse{P}}}{\RAcomplement{Q}}}}$}
			$\mIff$ \com{Proposition~\ref{prop:relationproperties}(vii)}
			\pred{$Q = \RAcompose{P}{\RAcomplement{\RAcompose{\RAconverse{P}}{\RAcomplement{Q}}}} \STleq \RAcompose{\RAcompose{P}{M}}{\RAcomplement{\RAcompose{\RAconverse{(\RAcompose{P}{M})}}{\RAcomplement{Q}}}}$}
			$\mImp$ \com{Transitivity of $\STleq$}
			\pred{$Q \STleq \RAcompose{\RAcompose{P}{M}}{\RAcomplement{\RAcompose{\RAconverse{(\RAcompose{P}{M})}}{\RAcomplement{Q}}}}$}
			$\mIff$ \com{Definition~\ref{def:residue}(i) \hsep Proposition~\ref{prop:residueproperties}(vii)}
			\pred{$Q \STleq \RAcompose{\RAcompose{P}{M}}{\big(\RArightresidual{(\RAcompose{P}{M})}{Q}\big)} \STleq Q$}
			$\mIff$ \com{Antisymmetry}
			\pred{$Q = \RAcompose{\RAcompose{P}{M}}{\big(\RArightresidual{(\RAcompose{P}{M})}{Q}\big)}$}
			$\mIff$ \com{Proposition~\ref{prop:test}}
			\pred{$\lnotation{\exists}{Y}{}{\RAcompose{\RAcompose{P}{M}}{Y} = Q}$}
		\Endproof

	\end{proof}

\end{corollary}
