\begin{corollary}
\label{cor:compute}
	Let $P$ be the relation containing confidential information. Let $Q$ be a relation representing an information observed on the monitored communication channel. Let $R$ be a filtering relation allowing for the selection of particular elements of the relations $P$ and $Q$. The confidential information included in $P$ is being leaked as that represented by $Q$ via the abstraction relation 
	\begin{enumerate}[(i)]
		\item $X = R \STmeet \RAconverse{(\RArightresidual{Q}{P})}$ if and only if $P \STleq \RAcompose{Q}{\big(\RAconverse{R} \STmeet (\RArightresidual{Q}{P})\big)}$
		\item $X = R \STmeet (\RArightresidual{P}{Q})$ if and only if $Q \STleq \RAcompose{P}{\big(R \STmeet (\RArightresidual{P}{Q})\big)}$
		\item $X = R \STmeet \RAsyq{P}{Q}$ if and only if $P \STleq \RAcompose{Q}{\big(\RAconverse{R} \STmeet (\RArightresidual{Q}{P})\big)} \nAnd Q \STleq \RAcompose{P}{\big(R \STmeet (\RArightresidual{P}{Q})\big)}$
	\end{enumerate}
	
\begin{proof}~
	\begin{enumerate}[(i)]
		\item 
			\Beginproof
				\pred{$P \STleq \RAcompose{Q}{\big(\RAconverse{R} \STmeet (\RArightresidual{Q}{P})\big)}$}
				$\mIff$ \com{Converse both sides \!\!\!\!\!\hsep\!\!\!\!\! Proposition~\ref{prop:relationproperties}(ii) \!\!\!\!\!\hsep\!\!\!\!\! Proposition~\ref{prop:relationproperties}(vi) \!\!\!\!\!\hsep\!\!\!\!\! Proposition~\ref{prop:relationproperties}(vii)}
				\pred{$\RAconverse{P} \STleq \RAcompose{\big(R \STmeet \RAconverse{(\RArightresidual{Q}{P})}\big)}{\RAconverse{Q}}$}
				$\mIff$ \com{Proposition~\ref{prop:compute}}
				\pred{$\lnotation{\exists}{X}{X = R \STmeet \RAconverse{(\RArightresidual{Q}{P})}}{\RAcompose{X}{\RAconverse{Q}} = \RAconverse{P}}$}
			\Endproof
		\item 
			\Beginproof
				\pred{$Q \STleq \RAcompose{P}{\big(R \STmeet (\RArightresidual{P}{Q})\big)}$}
				$\mIff$ \com{Converse both sides \!\!\!\!\!\hsep\!\!\!\!\! Proposition~\ref{prop:relationproperties}(ii) \!\!\!\!\!\hsep\!\!\!\!\! Proposition~\ref{prop:relationproperties}(vi) \!\!\!\!\!\hsep\!\!\!\!\! Proposition~\ref{prop:relationproperties}(vii)}
				\pred{$\RAconverse{Q} \STleq \RAcompose{\big(\RAconverse{R} \STmeet \RAconverse{(\RArightresidual{P}{Q})}\big)}{\RAconverse{P}}$}
				$\mIff$ \com{Proposition~\ref{prop:compute}}
				\pred{$\lnotation{\exists}{X}{X = R \STmeet (\RArightresidual{P}{Q})}{\RAcompose{\RAconverse{X}}{\RAconverse{P}} = \RAconverse{Q}}$}
			\Endproof
		\item
			\Beginproof
				\pred{$P \STleq \RAcompose{Q}{\big(\RAconverse{R} \STmeet (\RArightresidual{Q}{P})\big)} \nAnd Q \STleq \RAcompose{P}{\big(R \STmeet (\RArightresidual{P}{Q})\big)}$}
				$\mIff$ \com{Converse both sides \!\!\!\!\!\hsep\!\!\!\!\! Proposition~\ref{prop:relationproperties}(ii) \!\!\!\!\!\hsep\!\!\!\!\! Proposition~\ref{prop:relationproperties}(vi) \!\!\!\!\!\hsep\!\!\!\!\! Proposition~\ref{prop:relationproperties}(vii) \!\!\!\!\!\hsep\!\!\!\!\! Proposition~\ref{prop:residueproperties}(ii)}
				\pred{$\RAconverse{P} \STleq \RAcompose{\big(R \STmeet (\RAleftresidual{\RAconverse{P}}{\RAconverse{Q}})\big)}{\RAconverse{Q}} \nAnd \RAconverse{Q} \STleq \RAcompose{\big(\RAconverse{R} \STmeet (\RAleftresidual{\RAconverse{Q}}{\RAconverse{P}})\big)}{\RAconverse{P}}$}
				$\mIff$ \com{Proposition~\ref{prop:compute}}
				\pred{$\lnotation{\exists}{X}{\!\!X = R \!\STmeet\! (\RAleftresidual{\RAconverse{P}}{\RAconverse{Q}}) \!\nAnd\! \RAconverse{X} = \RAconverse{R} \!\STmeet\! (\RAleftresidual{\RAconverse{Q}}{\RAconverse{P}})}{\RAcompose{X}{\RAconverse{Q}} = \RAconverse{P} \!\nAnd\! \RAcompose{\RAconverse{X}}{\RAconverse{P}} = \RAconverse{Q}}$}
				$\mIff$ \com{Converse both sides \hsep Proposition~\ref{prop:relationproperties}(ii)}
				\pred{$\lnotation{\exists}{X}{X = R \!\STmeet\! (\RAleftresidual{\RAconverse{P}}{\RAconverse{Q}}) \!\nAnd\! X = R \!\STmeet\! (\RArightresidual{P}{Q})}{\RAcompose{X}{\RAconverse{Q}} = \RAconverse{P} \!\nAnd\! \RAcompose{\RAconverse{X}}{\RAconverse{P}} = \RAconverse{Q}}$}
				$\mIff$ \com{Golden Rule Axiom: $X = P \nAnd X = Q \mIff X = (P \STjoin Q) \nAnd \\ X = (P \STmeet Q)$}
				\pred{$\lnotation{\exists}{X}{X = \big[\big(R \STmeet (\RAleftresidual{\RAconverse{P}}{\RAconverse{Q}})\big) \STmeet \big(R \STmeet (\RArightresidual{P}{Q})\big)\big] \nAnd X = \big[\big(R \STmeet (\RAleftresidual{\RAconverse{P}}{\RAconverse{Q}})\big) \STjoin \big(R \STmeet (\RArightresidual{P}{Q})\big)\big]}{\RAcompose{X}{\RAconverse{Q}} = \RAconverse{P} \nAnd \RAcompose{\RAconverse{X}}{\RAconverse{P}} = \RAconverse{Q}}$}
				$\mIff$ \com{Distributivity of $\STmeet$ of $\STjoin$}
				\pred{$\lnotation{\exists}{X}{X = R \STmeet \big((\RAleftresidual{\RAconverse{P}}{\RAconverse{Q}}) \STmeet (\RArightresidual{P}{Q})\big) \nAnd X = R \STmeet \big((\RAleftresidual{\RAconverse{P}}{\RAconverse{Q}}) \STjoin (\RArightresidual{P}{Q})\big)}{\RAcompose{X}{\RAconverse{Q}} = \RAconverse{P} \nAnd \RAcompose{\RAconverse{X}}{\RAconverse{P}} = \RAconverse{Q}}$}
				$\mIff$ \com{Golden Rule Axiom: $X = R \STmeet (P \STmeet Q) \nAnd X = R \STmeet (P \STjoin Q) \mIff \\ X = R \STmeet P \STmeet Q \nAnd (P \STmeet Q) = (P \STjoin Q)$}
				\pred{$\lnotation{\exists}{X}{X = R \STmeet \big((\RAleftresidual{\RAconverse{P}}{\RAconverse{Q}}) \STmeet (\RArightresidual{P}{Q})\big) \nAnd \big[\big((\RArightresidual{P}{Q}) \STmeet (\RAleftresidual{\RAconverse{P}}{\RAconverse{Q}})\big) = \big((\RArightresidual{P}{Q}) \STjoin (\RAleftresidual{\RAconverse{P}}{\RAconverse{Q}})\big)\big]}{\RAcompose{X}{\RAconverse{Q}} = \RAconverse{P} \nAnd \RAcompose{\RAconverse{X}}{\RAconverse{P}} = \RAconverse{Q}}$}
				$\mIff$ \com{Proposition~\ref{prop:residueproperties}(ii)}
				\pred{$\lnotation{\exists}{X}{X = R \STmeet \big((\RAleftresidual{\RAconverse{P}}{\RAconverse{Q}}) \STmeet (\RArightresidual{P}{Q})\big) \nAnd \true}{\RAcompose{X}{\RAconverse{Q}} = \RAconverse{P} \nAnd \RAcompose{\RAconverse{X}}{\RAconverse{P}} = \RAconverse{Q}}$}
				$\mIff$ \com{Definition~\ref{def:symmetricquotient} \hsep Identity of $\AAnd$}
				\pred{$\lnotation{\exists}{X}{X = R \STmeet \RAsyq{P}{Q}}{\RAcompose{X}{\RAconverse{Q}} = \RAconverse{P} \nAnd \RAcompose{\RAconverse{X}}{\RAconverse{P}} = \RAconverse{Q}}$}
			\Endproof
	\end{enumerate}
	\vspace{-0.25in}
\end{proof}

\end{corollary}
