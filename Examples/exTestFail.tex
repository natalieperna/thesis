\begin{example}~
\label{ex:test_fail}
	Consider a case where the set of confidential information is represented as $P = \set{(1,3),(2,1),(3,4),(4,1),(5,5),(6,9)}$. Suppose that \AgentOne sends the information $Q = \set{(1,12),(2,1),(3,16),(4,12),(5,17),(6,18)}$ on the communication channel. We verify the existence of an abstraction relation by applying Corollary~\ref{cor:test}. 
	
	\Beginproof
		\pred{$P = \RAcompose{Q}{(\RArightresidual{Q}{P})} \Ors Q = \RAcompose{P}{(\RArightresidual{P}{Q})}$}
		$\mIff$ \com{Substitution: $P = \set{(1,3),(2,1),(3,4),(4,1),(5,5),(6,9)}$ and \\$Q = \set{(1,12),(2,1),(3,16),(4,12),(5,17),(6,18)}$ \hsep \\ Computation of $\RAcompose{Q}{(\RArightresidual{Q}{P})}$ and $\RAcompose{P}{(\RArightresidual{P}{Q})}$}
		\pred{$\set{(1,3),(2,1),(3,4),(4,1),(5,5),(6,9)} = \set{(2,1),(3,4),(5,5),(6,9)} \Ors \set{(1,12),(2,1),(3,16),(4,12),(5,17),(6,18)} = \set{(1,12),(3,16),(5,17),(6,18)}$}
		$\mIff$ \com{Equality}
		\pred{$\false \Ors \false$}
		$\mIff$ \com{Idempotency of $\Or$}
		\pred{$\false$}
	\Endproof
	\vspace{-0.65in}

\end{example}
