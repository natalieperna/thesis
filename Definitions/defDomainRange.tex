\begin{definition}
\label{def:domain_range}	
	Let $R \STleq X \times Y$ be a relation:
	\begin{enumerate}[(i)]
   		\item The \emph{domain}\index{domain} of the relation $R$ is given by
			\begin{equation*}
				\dom{R} = \sets{x}{\lnotation{\exists}{y}{y \in Y}{(x,y) \in R}}
			\end{equation*}
		\item The \emph{range}\index{range} of the relation $R$ is given by
			\begin{equation*}
				\ran{R} = \sets{y}{\lnotation{\exists}{x}{x \in X}{(x,y) \in R}}
			\end{equation*}
	\end{enumerate}
\end{definition}
