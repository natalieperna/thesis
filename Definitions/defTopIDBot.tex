\begin{definition}
\label{def:topidbot}~ 
	\begin{enumerate}[(i)]
		\item For every set $A$, the relation $\RAid$ on $A \times A$ is called the identity\index{relation!identity} relation on $A$ and is defined by
			\begin{equation*}
				\RAid \deq \sets{(x,x)}{x \in A}
			\end{equation*}
		\item For every two sets $A$ and $B$, the relation $\RAtop$ on $A \times B$ is called the universal relation\index{relation!universal} and is defined by
			\begin{equation*}
				\RAtop \deq \sets{(x,y)}{\true} 
			\end{equation*}
		\item For every two sets $A$ and $B$, the relation $\RAbot$ is called the empty relation\index{relation!empty} and is defined by
			\begin{equation*}
				\RAbot \deq \sets{(x,y)}{\false}
			\end{equation*}
	\end{enumerate}
\end{definition}
