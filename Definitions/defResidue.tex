\begin{definition}
\label{def:residue}
	Let $P$ and $Q$ be two relations. 
	\begin{enumerate}[(i)]
   		\item $\RArightresidual{P}{Q} \deq \RAcomplement{\RAcompose{\RAconverse{P}}{\RAcomplement{Q}}}$ is said to be the \textit{right residue}\index{residue!right residue} of $Q$ by $P$;
   		\item $\RAleftresidual{Q}{P}  \deq \RAcomplement{\RAcompose{\RAcomplement{Q}}{\RAconverse{P}}}$ is said to be the \textit{left residue}\index{residue!left residue} of $Q$ by $P$.
	\end{enumerate}
\end{definition}
