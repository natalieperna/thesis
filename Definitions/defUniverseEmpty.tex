\begin{definition}
\label{def:universeempty}~ 
	\begin{enumerate}[(i)]
		\item The universal set\index{set!universal}, denoted by $\STtop$, is the set fixed within the framework of a theory and consisting of all objects considered in this theory.
		\item The set $\STbot$ is called the empty set\index{set!empty} and is defined by
			\begin{equation*}
				\STbot \deq \sets{x}{\false}
			\end{equation*}
	\end{enumerate}
\end{definition}
