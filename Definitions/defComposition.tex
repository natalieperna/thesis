\begin{definition}
\label{def:composition}
	Let $R \STleq X \times Y$ and $S \STleq Y \times Z$ be relations. Then, their composition\index{composition} $\RAcompose{R}{S}$ is defined by
	\begin{equation*}
		\RAcompose{R}{S} = \sets{(x,z)}{\lnotation{\exists}{y}{y \in Y}{(x,y) \in R \nAnd (y,z) \in S}}
	\end{equation*}
\end{definition}
